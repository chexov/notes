osx: javascript lint + vim

June 5 2011, 8:33 AM  by Anton Checkhov

brew install jsl
jsl -help:conf > ~/jsl.conf

cat ~/.vim/ftplugin/javascript.vim
set makeprg=jsl\ -nologo\ -nofilelisting\ -nosummary\ -nocontext\ -conf\ '~/jsl.conf'\ -process\ %
set errorformat=%f(%l):\ %m
"make F10 call make for linting etc.
inoremap :make
map :make
EOF
